/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
struct ListNode* reverseList(struct ListNode* head) {
    struct ListNode *prev,*curr,*link;
    prev=NULL;
    curr=head;
    while(curr)
    {
        link=curr->next;
        curr->next=prev;
        prev=curr;
        curr=link;
    }
    head=prev;
    return head;
}
