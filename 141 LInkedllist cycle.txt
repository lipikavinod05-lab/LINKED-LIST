bool hasCycle(struct ListNode *head) {
    
        struct ListNode *slow;
        struct ListNode *fast;
        slow=head;
        fast=head;
        while(fast!=NULL && fast->next!=NULL)
        {
            slow=slow->next;
            fast=fast->next;
            fast=fast->next;
            if(slow==fast)
            {
                return true;
            }
        }
        return false;
    }
